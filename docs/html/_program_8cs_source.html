141  HandleCommandAsync(SocketMessage _messageParam)
142  {
143  // Don't process the command if it was a System Message.
144  if (!(_messageParam is SocketUserMessage message)) {
145  return;
146  }
147 
148  //Create a number to track where the prefix ends and the command begins.
149  int argPos = 0;
150 
151  //Create a Command Context.
152  var context = new SocketCommandContext(g_client, message);
153 
154  //Create prefix string.
155  string prefix = "";
156 
157  //On database file exists.
158  if (File.Exists($"Databases/{context.Guild.Id}.db")) {
159  //Get the prefix from the guild settings.
160  prefix = Modules.CoreModule.GetPrefix(context.Guild.Id);
161  }
162 
163  //Determine if the message is a command with prefix and if the author is not a bot.
164  if (!(message.HasStringPrefix(prefix, ref argPos) ||
165  message.HasCharPrefix('~', ref argPos) ||
166  message.HasMentionPrefix(g_client.CurrentUser, ref argPos)) ||
167  message.Author.IsBot) {
168  return;
169  }
170 
171  //Execute the command. (result does not indicate a return value,
172  //rather an object stating if the command executed successfully)
173  var result = await g_commands.ExecuteAsync(context, argPos, g_services);
174  if (!result.IsSuccess) {
175  //Create embed object
176  var embed = Modules.CoreModule.SimpleEmbed(Color.Red, "Error", result.ErrorReason);
177 
178  //Reply embed
179  await context.Channel.SendMessageAsync("", false, embed.Build());
180  }
